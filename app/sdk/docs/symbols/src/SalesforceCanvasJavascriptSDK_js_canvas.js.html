186             /**
187             * @description Calls a defined function for each element in an object.
188             * @param {Object} obj The object to loop through.  
189                 The object can be an array, an array like object, or a map of properties.
190             * @param {Function} it The callback function to run for each element
191             * @param {Object} [ctx] The context object to be used for the callback function.
192                 Defaults to the original object if not provided.
193             */
194             each: function (obj, it, ctx) {
195                 if ($.isNil(obj)) {
196                     return;
197                 }
198                 var nativ = aproto.forEach, i = 0, l, key;
199                 l = obj.length;
200                 ctx = ctx || obj;
201                 // @todo: looks like native method will not break on return false; maybe throw breaker {}
202                 if (nativ && nativ === obj.forEach) {
203                     obj.forEach(it, ctx);
204                 }
205                 else if ($.isNumber(l)) { // obj is an array-like object
206                     while (i < l) {
207                         if (it.call(ctx, obj[i], i, obj) === false) {
208                             return;
209                         }
210                         i += 1;
211                     }
212                 }
213                 else {
214                     for (key in obj) {
215                         if ($.hasOwn(obj, key) && it.call(ctx, obj[key], key, obj) === false) {
216                             return;
217                         }
218                     }
219                 }

333             /**
334              * @description Serializes an object into a string that can be used as a URL query string.
335              * @param {Object|Array} a The array or object to serialize
336              * @param {Boolean} [encode=false] Indicates that the string should be encoded
337              * @returns {String} A string representing the object as a URL query string.
338              */
339             param: function (a, encode) {
340                 var s = [];
341 
342                 encode = encode || false;
343 
344                 function add( key, value ) {
345 
346                     if ($.isNil(value)) {return;}
347                     value = $.isFunction(value) ? value() : value;
348                     if ($.isArray(value)) {
349                         $.each( value, function(v, n) {
350                             add( key, v );
351                         });
352                     }
353                     else {
354                         if (encode) {
355                             s[ s.length ] = encodeURIComponent(key) + "=" + encodeURIComponent(value);
356                         }
357                         else {
358                             s[ s.length ] = key + "=" + value;
359                         }
360                     }
361                 }
362 
363                 if ( $.isArray(a)) {
364                     $.each( a, function(v, n) {
365                         add( n, v );
366                     });
367                 } else {
368                     for ( var p in a ) {
369                         if ($.hasOwn(a, p)) {
370                             add( p, a[p]);
371                         }
372                     }
373                 }
374                 return s.join("&").replace(/%20/g, "+");
375             },

380              * passing in <code>?param=value1&​param=value2</code> will not return <code>['value1', 'value2']</code>
381              *
382              * @param {String} q ?param1=value1&param2=value2

457             /**
458              * @description Validates the event name.
459              * @param {String} name Name of the event; can include the namespace (namespace.name).
460              * @param {String} res Reserved namespace name to allow against default
461              * @returns {int} error code, 0 if valid
462              */
463              validEventName : function(name, res) {
464                 var ns, parts = name.split(/\./),
465                     regex = /^[$A-Z_][0-9A-Z_$]*$/i,
466                     reserved = {
467                         'sfdc':true, 'canvas':true,
468                         'force':true, 'salesforce':true, 'chatter':true
469                     };
470                 $.each($.isArray(res) ? res : [res], function (v) {
471                     reserved[v] = false;
472                 });
473                 if (parts.length > 2) {
474                     return 1;
475                 }
476                 if (parts.length === 2) {
477                     ns = parts[0].toLowerCase();
478                     if (reserved[ns]) {
479                         return 2;
480                     }
481                 }
482                 if (!regex.test(parts[0]) || !regex.test(parts[1])) {
483                     return 3;
484                 }
485                 return 0;

512             /**
513             * @description Adds a module to the global.Sfdc.canvas object.
514             * @param {String} ns The namespace for the new module
515             * @decl {Object} The module to add.
516             * @returns {Object} The global.Sfdc.canvas object with a new module added.

518             module: function(ns, decl) {
519                 var parts = ns.split('.'), parent = global.Sfdc.canvas, i, length;
520 
521                 // strip redundant leading global
522                 if (parts[1] === 'canvas') {
523                     parts = parts.slice(2);
524                 }
525 
526                 length = parts.length;
527                 for (i = 0; i < length; i += 1) {

531                     }
532                     parent = parent[parts[i]];
533                 }

536                     decl = decl();
537                 }
538                 return $.extend(parent, decl);

763                 } catch(e) {}

765                 // IE, the document is not inside a frame

774                     }
775                 });
